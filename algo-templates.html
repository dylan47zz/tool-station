🪟 滑动窗口
在序列上找一个满足约束的连续子区间

🎯 第一性原理

滑动窗口要解决的问题本质上只有一个：在序列上找一个满足约束的连续子区间。约束分两类：找满足条件的最短子区间（最小窗口），找不违反条件的最长子区间（最大窗口）。所有变种的结构不变，变的只有三处：收缩条件、答案更新位置、window 数据结构。

📍 最小窗口模板
找满足约束的最短子区间。核心：满足条件时贪心左缩。
Python 最小窗口 · 通用 
def minWindow(s, need):
    from collections import Counter
    window = Counter()
    valid = 0  # 已满足的字符种类数

    left = right = 0
    min_len = float('inf')
    start = 0

    while right < len(s):
        c = s[right]
        right += 1
        if c in need:
            window[c] += 1
            if window[c] == need[c]:
                valid += 1

        # 满足约束 → 左缩贪心
        while valid == len(need):
            if right - left < min_len:
                min_len = right - left
                start = left
            d = s[left]
            left += 1
            if d in need:
                if window[d] == need[d]:
                    valid -= 1
                window[d] -= 1

    return "" if min_len == float('inf') else s[start:start+min_len]
⚡ valid 计数器

用 valid 记录已满足的字符种类数，判断覆盖从 O(26) 降到 O(1)

⚡ 只记 need 中的字符

window 只记录 need 中出现的字符，减少无效操作

def minSubArrayLen(target, nums):
    left = right = 0
    window_sum = 0
    min_len = float('inf')

    while right < len(nums):
        window_sum += nums[right]
        right += 1

        while window_sum >= target:
            min_len = min(min_len, right - left)
            window_sum -= nums[left]
            left += 1

    return 0 if min_len == float('inf') else min_len

def maxWindow(s):
    from collections import Counter
    window = Counter()
    left = right = 0
    max_len = 0

    while right < len(s):
        c = s[right]
        right += 1
        window[c] += 1

        # 违反约束 → 左缩恢复
        while 违反条件:
            d = s[left]
            left += 1
            window[d] -= 1

        # 缩完一定合法，更新答案
        max_len = max(max_len, right - left)

    return max_len

def lengthOfLongestSubstring(s):
    from collections import Counter
    window = Counter()
    left = right = 0
    max_len = 0

    while right < len(s):

        right += 1

def characterReplacement(s, k):
    from collections import Counter
    window = Counter()
    left = right = 0
    max_len = 0
    max_count = 0

    while right < len(s):
        c = s[right]
        right += 1
        window[c] += 1
        max_count = max(max_count, window[c])

        # 窗口长度 - 最高频 > k → 需替换次数超限
        while right - left - max_count > k:
            d = s[left]
            left += 1
            window[d] -= 1

        max_len = max(max_len, right - left)

    return max_len
💡 max_count 可以不缩小！只影响答案上界，更小的值只会让窗口更早收缩，答案只会更小或不变。O(26n) → O(n)

📍 计数窗口模板

Python 计数窗口 · 通用 
def countSubarrays(nums, k):
    return atMost(nums, k)  # 恰好K = atMost(K) - atMost(K-1)

def atMost(nums, k):
    from collections import Counter
    window = Counter()
    left = right = 0
    count = 0

    while right < len(nums):
        c = nums[right]
        right += 1
        window[c] += 1

        while len(window) > k:  # 超过k种 → 违反
            d = nums[left]
            left += 1
            window[d] -= 1
            if window[d] == 0:
                del window[d]

        # 以 right-1 结尾的所有合法子数组个数
        count += right - left

    return count

🎯 right - left 的含义

以 right-1 为右端点的合法子数组恰好有 right - left 个（左端点可以从 left 到 right-1）。
例：[a b c d e]，left=2, right=5 → 以 e 结尾的合法子数组：cde, bcde, abcde = 3 = right-left

def numSubarrayProductLessThanK(nums, k):
    if k <= 1: return 0
    left = right = 0
    prod = 1
    count = 0

    while right < len(nums):
        prod *= nums[right]
        right += 1

        while prod >= k:
            prod //= nums[left]
            left += 1

        count += right - left

    return count

def fixedWindow(s, k):
    from collections import Counter
    window = Counter()
    # 初始化前 k 个元素
    for i in range(k):
        window[s[i]] += 1
    if 满足条件: 记录答案

    # 滑动
    for i in range(k, len(s)):
        window[s[i]] += 1         # 右端进入
        window[s[i - k]] -= 1      # 左端离开
        if window[s[i - k]] == 0:
            del window[s[i - k]]
        if 满足条件: 记录答案

def checkInclusion(s1, s2):
    from collections import Counter
    need = Counter(s1)
    window = Counter()
    valid = 0
    k = len(s1)

    for i in range(len(s2)):
        c = s2[i]
        if c in need:
            window[c] += 1
            if window[c] == need[c]: valid += 1

        if i >= k:
            d = s2[i - k]
            if d in need:
                if window[d] == need[d]: valid -= 1
                window[d] -= 1

        if valid == len(need):
            return True

    return False

from collections import deque

def maxSlidingWindow(nums, k):
    q = deque()  # 存下标，维护单调递减
    res = []

    for i in range(len(nums)):
        while q and nums[q[-1]] <= nums[i]:
            q.pop()
        q.append(i)

        if q[0] <= i - k:
            q.popleft()

        if i >= k - 1:
            res.append(nums[q[0]])

    return res

🔍 二分搜索
在有序空间上找分界点。核心：闭区间 [left, right]，循环不变量。

🎯 第一性原理

二分搜索的本质：在有序空间上找一个分界点，使得分界点左侧满足某性质、右侧不满足（或反过来）。所有变种的区别仅在于"找什么分界点"——左边界、右边界、还是任意一个。

Python · 统一框架 闭区间 [left, right] 不变量 
def binarySearch(nums, target):
    left, right = 0, len(nums) - 1

    while left <= right:  # 闭区间，= 时还有一个元素
        mid = left + (right - left) // 2

        if nums[mid] == target:
            # 标准二分：直接返回
            return mid
            # 找左边界：继续向左收缩
            # right = mid - 1
            # 找右边界：继续向右收缩
            # left = mid + 1
        elif nums[mid] < target:
            left = mid + 1
        else:
            right = mid - 1

    return -1

def leftBound(nums, target):
    left, right = 0, len(nums) - 1

    while left <= right:
        mid = left + (right - left) // 2

        if nums[mid] < target:
            left = mid + 1
        elif nums[mid] > target:
            right = mid - 1
        else:
            right = mid - 1  # 命中但继续向左收缩

    # left 指向第一个 >= target 的位置
    if left >= len(nums) or nums[left] != target:
        return -1
    return left

def minSatisfying(low, high, condition):
    left, right = low, high
    while left <= right:
        mid = left + (right - left) // 2
        if condition(mid):  # mid 能满足 → 尝试更小
            right = mid - 1
        else:                 # mid 不满足 → 需要更大
            left = mid + 1
    return left

🌳 回溯算法
在决策树上做 DFS + 剪枝。核心：选或不选，走到底回退。

🎯 第一性原理

回溯的本质：在决策树上做深度优先遍历。每个节点代表一个"选择"，每条边代表"做了某个选择"。走到叶子节点就是一条完整决策路径，回退就是撤销选择。所有变种的区别只有三处：选择列表、剪枝条件、终止条件。

Python · 统一框架 所有回溯题的骨架 
def backtrack(路径, 选择列表):
    if 满足结束条件:
        result.append(路径.copy())
        return

    for 选择 in 选择列表:
        if 剪枝条件: continue
        做选择       # 路径.add(选择)
        backtrack(路径, 新选择列表)
        撤销选择     # 路径.remove(选择)
类型选择列表剪枝终止条件
子集i → n无（全选）每层都收集
组合i → n剩余不够 k 个路径长度 == k
排列0 → n（未使用的）used[i]路径长度 == n
分割start → n子串不合法start == n

def subsets(nums):
    res = []
    def backtrack(start, path):
        res.append(path[:])  # 每层都收集
        for i in range(start, len(nums)):
            # 有重复元素时剪枝：
            # if i > start and nums[i] == nums[i-1]: continue
            path.append(nums[i])
            backtrack(i + 1, path)
            path.pop()
    backtrack(0, [])
    return res

def combine(n, k):
    res = []
    def backtrack(start, path):
        if len(path) == k:
            res.append(path[:])
            return
        # 剪枝：剩余元素不够 k - len(path) 个
        for i in range(start, n - (k - len(path)) + 2):
            path.append(i)
            backtrack(i + 1, path)
            path.pop()
    backtrack(1, [])
    return res

def permute(nums):
    res = []
    used = [False] * len(nums)

    def backtrack(path):
        if len(path) == len(nums):
            res.append(path[:])
            return
        for i in range(len(nums)):
            if used[i]: continue
            # 有重复元素时：
            # if i > 0 and nums[i] == nums[i-1] and not used[i-1]: continue
            used[i] = True
            path.append(nums[i])
            backtrack(path)
            path.pop()
            used[i] = False

    backtrack([])
    return res

def partition(s):

        if start == len(s):

📊 动态规划
找状态定义和状态转移方程。核心：最优子结构 + 无后效性。

🎯 第一性原理

DP 的本质：用子问题的最优解组合出原问题的最优解。拆解为三步：(1) 定义 dp[i] 或 dp[i][j] 的含义；(2) 推导状态转移方程；(3) 确定边界条件和遍历顺序。所有 DP 题的难点只在第 (2) 步——怎么从子问题推出当前问题。

# 3. 返回答案

# O(n²) DP
def lengthOfLIS(nums):
    dp = [1] * len(nums)  # dp[i] = 以 nums[i] 结尾的 LIS 长度
    for i in range(len(nums)):
        for j in range(i):
            if nums[j] < nums[i]:
                dp[i] = max(dp[i], dp[j] + 1)
    return max(dp)

# O(n log n) 贪心 + 二分
def lengthOfLIS(nums):
    tails = []  # tails[i] = 长度为 i+1 的 LIS 的最小末尾
    for x in nums:
        pos = bisect_left(tails, x)
        if pos == len(tails):
            tails.append(x)
        else:
            tails[pos] = x
    return len(tails)

📍 最长公共子序列 (LCS)
Python 二维 DP 
def longestCommonSubsequence(text1, text2):
    m, n = len(text1), len(text2)
    # dp[i][j] = text1[:i] 和 text2[:j] 的 LCS 长度
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if text1[i-1] == text2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])

    return dp[m][n]

def minDistance(word1, word2):
    m, n = len(word1), len(word2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]
    for i in range(m + 1): dp[i][0] = i
    for j in range(n + 1): dp[0][j] = j

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if word1[i-1] == word2[j-1]:
                dp[i][j] = dp[i-1][j-1]
            else:
                dp[i][j] = min(
                    dp[i-1][j] + 1,    # 删
                    dp[i][j-1] + 1,    # 插
                    dp[i-1][j-1] + 1  # 替
                )
    return dp[m][n]

# 0-1 背包：每个物品最多选 1 次
def knapsack01(W, wt, val):
    dp = [0] * (W + 1)
    for i in range(len(wt)):
        for j in range(W, wt[i] - 1, -1):  # ← 逆序！保证每物品只用1次
            dp[j] = max(dp[j], dp[j - wt[i]] + val[i])
    return dp[W]

# 完全背包：每个物品可选无限次
def knapsackComplete(W, wt, val):
    dp = [0] * (W + 1)
    for i in range(len(wt)):
        for j in range(wt[i], W + 1):  # ← 正序！允许重复选
            dp[j] = max(dp[j], dp[j - wt[i]] + val[i])
    return dp[W]

    dp[0] = 1

        for j in range(coin, amount + 1):  # 内层正序 → 完全背包

def maxProduct(nums):
    cur_max = cur_min = res = nums[0]
    for x in nums[1:]:
        candidates = (x, cur_max * x, cur_min * x)
        cur_max, cur_min = max(candidates), min(candidates)
        res = max(res, cur_max)
    return res

# 最长回文子串 (LC 5) — 中心扩展
def longestPalindrome(s):
    res = ""
    def expand(l, r):
        while l >= 0 and r < len(s) and s[l] == s[r]:
            l -= 1; r += 1
        return s[l+1:r]
    for i in range(len(s)):
        res = max(res, expand(i, i), expand(i, i+1), key=len)
    return res

def canFinish(numCourses, prerequisites):
    from collections import defaultdict, deque
    graph = defaultdict(list)
    indegree = [0] * numCourses
    for a, b in prerequisites:
        graph[b].append(a)
        indegree[a] += 1

    q = deque([i for i in range(numCourses) if indegree[i] == 0])
    count = 0
    while q:
        node = q.popleft()
        count += 1
        for nb in graph[node]:
            indegree[nb] -= 1
            if indegree[nb] == 0:
                q.append(nb)
    return count == numCourses

def nextGreaterElement(nums):
    res = [-1] * len(nums)
    stack = []  # 存下标，维护单调递减栈

    for i in range(len(nums)):
        while stack and nums[stack[-1]] < nums[i]:
            res[stack.pop()] = nums[i]  # 弹出的元素找到了下一个更大
        stack.append(i)

    return res

def trap(height):
    stack = []
    res = 0
    for i in range(len(height)):
        while stack and height[stack[-1]] <= height[i]:
            mid = stack.pop()
            if stack:
                h = min(height[stack[-1]], height[i]) - height[mid]
                w = i - stack[-1] - 1
                res += h * w
        stack.append(i)
    return res

Python 带路径压缩和按秩合并 
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n
        self.count = n  # 连通分量数

    def find(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])  # 路径压缩
        return self.parent[x]

    def union(self, x, y):
        px, py = self.find(x), self.find(y)
        if px == py: return
        # 按秩合并
        if self.rank[px] < self.rank[py]:
            px, py = py, px
        self.parent[py] = px
        if self.rank[px] == self.rank[py]:
            self.rank[px] += 1
        self.count -= 1

    def connected(self, x, y):
        return self.find(x) == self.find(y)

def numIslands(grid):
    if not grid: return 0
    m, n = len(grid), len(grid[0])
    uf = UnionFind(m * n + 1)  # +1 给水域虚拟节点
    dummy = m * n

    for i in range(m):
        for j in range(n):
            if grid[i][j] == '0':
                uf.union(i * n + j, dummy)
            else:
                for di, dj in [(0,1),(1,0)]:
                    ni, nj = i+di, j+dj
                    if 0 <= ni < m and 0 <= nj < n and grid[ni][nj] == '1':
                        uf.union(i*n+j, ni*n+nj)

    return uf.count - 1  # 减去虚拟节点
